172 std::pair<std::array<double, 3>, std::array<double, 3>> latchToGridSize(std::array<double, 3> boxMin,
173 std::array<double, 3> boxMax) {
174 for (size_t ind = 0; ind < 3; ++ind) {
175 double currentGridSize = (*_gridSize)[ind];
176 // For boxmin, the lower domain boundary is 0, so that's always fine!
177 boxMin[ind] = std::round(boxMin[ind] / currentGridSize) * currentGridSize;
178 // update boxmax only if it isn't at the very top of the domain!
179 if (boxMax[ind] != _domainLength[ind]) {
180 boxMax[ind] = std::round(boxMax[ind] / currentGridSize) * currentGridSize;
181 }
182 }
183 return {boxMin, boxMax};
184 }
