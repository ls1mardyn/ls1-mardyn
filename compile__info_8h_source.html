13int get_compiler_info(char *info_str) {
14 /* For compiler predfined macros used for identification see
15 * http://predef.sourceforge.net/precomp.html
16 */
17
18 /* Cray compiler */
19#if defined(_CRAYC)
20 sprintf( info_str, "Cray %d.%d", _RELEASE, _RELEASE_MINOR);
21
22 /* Intel compiler */
23#elif defined(__INTEL_COMPILER)
24 int version = (int) __INTEL_COMPILER / 100;
25 int revision = ((int) __INTEL_COMPILER % 100) / 10;
26 int patch = (int) __INTEL_COMPILER % 10;
27 sprintf(info_str, "Intel %d.%d.%d", version, revision, patch);
28
29 /* PGI compiler */
30#elif defined(__PGI)
31 int version = (int) __PGIC__;
32 int revision = (int) __PGIC_MINOR__;
33 int patch = (int) __PGIC_PATCHLEVEL__;
34 sprintf(info_str, "PGI %d.%d.%d", version, revision, patch);
35
36 /* Sun compiler */
37#elif defined(__SUNPRO_C)
38 int version = (__SUNPRO_C >> 12) & 0xf;
39 int revision_digit1 = (__SUNPRO_C >> 8) & 0xf;
40 int revision_digit2 = (__SUNPRO_C >> 4) & 0xf;
41 int patch = (__SUNPRO_C >> 0 ) & 0xf;
42 sprintf(info_str, "Sun %d.%d%d.%d", version, revision_digit1, revision_digit2, patch);
43
44#elif defined(_SX)
45 sprintf(info_str, "NEC SX, rev. %d", __SXCXX_REVISION);
46
47 /* Clang compiler */
48#elif defined(__clang__)
49 int version = __clang_major__;
50 int revision = __clang_minor__;
51 int patch = __clang_patchlevel__;
52 sprintf(info_str, "Clang %d.%d.%d", version, revision, patch);
53
54 /* GNU compiler */
55#elif defined(__GNUC__)
56# if defined(__GNUC_PATCHLEVEL__)
57 sprintf(info_str, "GNU %d.%d.%d", __GNUC__, __GNUC_MINOR__, __GNUC_PATCHLEVEL__);
58# else
59 sprintf(info_str, "GNU %d.%d", __GNUC__, __GNUC_MINOR__);
60# endif
61
62 /* unknown */
63#else
64 sprintf(info_str, "unknown");
65#endif
66
67 return 0;
68}
