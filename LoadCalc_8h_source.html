63 double getOwn(int index1, int index2) const override {
64 if (index2 < _count2 && index1 < _count1) {
65 return accessVec(_ownTime, index1, index2);
66 } else {
67 // extrapolation
68 return index1 * (index1 - 1) / 2 * _ownConst[0] + index2 * (index2 - 1) / 2 * _ownConst[1]
69 + index2 * index1 * _ownConst[2];
70 }
71 }

135 static double costsNeighbour(int index1, int index2, std::array<double, 3> consts) noexcept {
136 return index1 * index1 * consts[0] + index2 * index2 * consts[1] + index1 * index2 * 2 * consts[2];

144 void writeVec(std::ostream& out, const std::vector<double>& vec) const {
145 for (int index1 = 0; index1 < _count1; ++index1) {
146 for (int index2 = 0; index2 < _count2 - 1; ++index2) {
147 out << accessVec(vec, index1, index2) << ";";
148 }
149 //no terminating ; for the last entry
150 if (vec.size() != 0) {
151 out << accessVec(vec, index1, _count2 - 1);
152 }
153 out << std::endl;
154 }
