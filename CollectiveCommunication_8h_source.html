145 void allreduceCustom(ReduceType type) override {
146 Log::global_log->debug() << "CollectiveCommunication: custom Allreduce" << std::endl;
147#if ENABLE_AGGLOMERATED_REDUCE
148 setMPIType();
149 MPI_Op agglomeratedTypeAddOperator;
150 const int commutative = 1;
151 valType * startOfValues = _values.data();
152
153 switch (type) {
154 case ReduceType::SUM:
155 MPI_CHECK(
156 MPI_Op_create(
157 (MPI_User_function * ) CollectiveCommunication::add,
158 commutative, &agglomeratedTypeAddOperator));
159 break;
160 case ReduceType::MAX:
161 MPI_CHECK(
162 MPI_Op_create(
163 (MPI_User_function * ) CollectiveCommunication::max,
164 commutative, &agglomeratedTypeAddOperator));
165 break;
166 case ReduceType::MIN:
167 MPI_CHECK(
168 MPI_Op_create(
169 (MPI_User_function * ) CollectiveCommunication::min,
170 commutative, &agglomeratedTypeAddOperator));
171 break;
172 default:
173 std::ostringstream error_message;
174 error_message<<"invalid reducetype, aborting." << std::endl;
175 MARDYN_EXIT(error_message.str());
176 }
177
178 MPI_CHECK(
179 MPI_Allreduce(MPI_IN_PLACE, startOfValues, 1, _agglomeratedType, agglomeratedTypeAddOperator, _communicator));
180 MPI_CHECK(MPI_Op_free(&agglomeratedTypeAddOperator));
181 MPI_CHECK(MPI_Type_free(&_agglomeratedType));
182#else
183 for (unsigned int i = 0; i < _types.size(); i++) {
184 MPI_Op op = MPI_NO_OP;
185 switch(type) {
186 case ReduceType::SUM:
187 op = MPI_SUM;
188 break;
189 case ReduceType::MIN:
190 op = MPI_MIN;
191 break;
192 case ReduceType::MAX:
193 op = MPI_MAX;
194 break;
195 default:
196 std::ostringstream error_message;
197 error_message<<"invalid reducetype, aborting." << std::endl;
198 MARDYN_EXIT(error_message.str());
199 }
200 MPI_CHECK(MPI_Allreduce( MPI_IN_PLACE, &_values[i], 1, _types[i], op, _communicator ));
201 }
202#endif
203 }

246 void setMPIType() {
247 int numblocks = _values.size();
248 std::vector<int> blocklengths(numblocks);
249 std::vector<MPI_Aint> disps(numblocks);
250 int disp = 0;
251 for (int i = 0; i < numblocks; i++) {
252 blocklengths[i] = 1;
253 disps[i] = disp;
254 disp += sizeof(valType);
255 }
256 MPI_Datatype * startOfTypes = &(_types[0]);
257#if MPI_VERSION >= 2 && MPI_SUBVERSION >= 0
258 MPI_CHECK(
259 MPI_Type_create_struct(numblocks, blocklengths.data(), disps.data(),
260 startOfTypes, &_agglomeratedType));
261#else
262 MPI_CHECK( MPI_Type_struct(numblocks, blocklengths.data(), disps.data(), startOfTypes, &_agglomeratedType) );
263#endif
264 MPI_CHECK(MPI_Type_commit(&_agglomeratedType));
265 }

278 static void add(valType *invec, valType *inoutvec, int */*len*/,
279 MPI_Datatype *dtype) {
280 int numints;
281 int numaddr;
282 int numtypes;
283 int combiner;
284
285 MPI_CHECK(
286 MPI_Type_get_envelope(*dtype, &numints, &numaddr, &numtypes,
287 &combiner));
288
289 std::vector<int> arrayInts(numints);
290 std::vector<MPI_Aint> arrayAddr(numaddr);
291 std::vector<MPI_Datatype> arrayTypes(numtypes);
292
293 MPI_CHECK(
294 MPI_Type_get_contents(*dtype, numints, numaddr, numtypes,
295 arrayInts.data(), arrayAddr.data(), arrayTypes.data()));
296
297 for (int i = 0; i < numtypes; i++) {
298 if (arrayTypes[i] == MPI_INT) {
299 inoutvec[i].v_int += invec[i].v_int;
300 }
301 else if (arrayTypes[i] == MPI_UNSIGNED_LONG) {
302 inoutvec[i].v_unsLong += invec[i].v_unsLong;

304 else if (arrayTypes[i] == MPI_FLOAT) {
305 inoutvec[i].v_float += invec[i].v_float;
306 }
307 else if (arrayTypes[i] == MPI_DOUBLE) {
308 inoutvec[i].v_double += invec[i].v_double;
309 }
310 else if (arrayTypes[i] == MPI_LONG_DOUBLE) {
311 inoutvec[i].v_longDouble += invec[i].v_longDouble;
