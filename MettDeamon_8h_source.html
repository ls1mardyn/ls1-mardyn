438 class Bin
439 {
440 friend class BinQueue;
441 public:
442 Bin(const std::vector<Molecule>& vec, uint32_t nIndex) : _next(nullptr), _nIndex(nIndex)
443 {
444 for(const auto& p:vec)
445 _particles.push_back(p);
446 }
447 uint32_t getIndex() {return _nIndex;}

449 Bin* _next;
450 uint32_t _nIndex;
451 std::vector<Molecule> _particles;
452 };
453private:
454 Bin *_first, *_last, *_actual;
455 uint32_t _numBins;
456 uint32_t _nRoundCount;
457 uint64_t _numParticles;
458 uint64_t _maxID;
459
460
461public:
462 BinQueue() :
463 _first(nullptr),
464 _last(nullptr),
465 _actual(nullptr),
466 _numBins(0),
467 _nRoundCount(0),
468 _numParticles(0),
469 _maxID(0)
470 {
471 }
472
473 BinQueue(std::vector<Molecule> vec) :
474 _first(nullptr),
475 _last(nullptr),
476 _actual(nullptr),
477 _numBins(0),
478 _nRoundCount(0),
479 _numParticles(0),
480 _maxID(0)
481 {
482 enque(std::move(vec));
483 }
484
485 ~BinQueue() {
486 if(_last){
487 // break connection of last element to first element (established via connectTailToHead())
488 _last->_next = nullptr;
489 }
490 auto ptr = _first;
491 while (ptr) {
492 // as long as ptr isn't nullptr we continue to delete the next element.
493 auto ptr_next = ptr->_next;
494 delete ptr;
495 ptr = ptr_next;
496 }
497 }
498
499 bool isEmpty() {
500 return _first == nullptr;
501 }
502
503 void enque(std::vector<Molecule> vec)
504 {
505 Bin* ptr = new Bin(vec, _numBins);
506 if (isEmpty()) {
507 _last = ptr;
508 _first = ptr;
509 _actual = ptr;
510 } else {
511 _last->_next = ptr;
512 _last = ptr;
513 }
514 _last->_next = _first; // connect tail to head
515 _numBins++;
516 _numParticles += vec.size();
517 // update max particle ID
518 if(_numParticles > 0)
519 {
520 std::vector<Molecule>::iterator it;
521 if(vec.empty())
522 it = vec.end();
523 else if(vec.size() == 1)
524 it = vec.begin();
525 else
526 it = max_element(vec.begin(), vec.end(), molecule_id_compare);
527
528 if(it != vec.end() )
529 _maxID = ( (*it).getID()>_maxID ? (*it).getID() : _maxID);
530 }
531 else
532 _maxID = 0;
533 }
534
535 void deque()
536 {
537 if (isEmpty()) {
538 return;
539 }
540 else if (_first == _last) {
541 _numParticles -= _first->getNumParticles();
542 delete _first;
543 _first = _last = nullptr;
544 _numBins--;
545 }
546 else {
547 Bin* ptr = _first;
548 while(ptr->_next != _last) {
549 ptr = ptr->_next;
550 }
551 _numParticles -= ptr->_next->getNumParticles();
552 delete ptr->_next;
553 ptr->_next = nullptr;
554 _last = ptr;
555 _numBins--;
556 _last->_next = _first; // connect tail to head
557 }
558 }
559
560 void clear()
561 {
562 while (!isEmpty()) {
563 deque();
564 }
565 }
566
567 std::vector<Molecule>& head() {
568 return _first->_particles;
569 }

572 return _actual->_particles;
573 }

581 void connectTailToHead()
582 {
583 _last->_next=_first;
584 _actual = _first;
585 }
586
587 bool next()
588 {
589 _actual = _actual->_next;
590 if(_actual == _first)
591 _nRoundCount++;
592 bool bSuccess = dynamic_cast<Bin*>(_actual);
593 return bSuccess;
594 }
595
596 bool activateBin(uint32_t nBinIndex)
597 {
598 Bin* ptr = _first;
599 while(ptr != nullptr)
600 {
601 if(ptr->_nIndex == nBinIndex) {
602 _actual = ptr;
603 return true;
604 }
605 ptr = ptr->_next;
606 if(ptr == _first)
607 break;
608 }
609 return false;
610 }

618private:
619 static bool molecule_id_compare(Molecule a, Molecule b)
620 {
621 return (a.getID() < b.getID());
622 }
623 void determineMaxID()
624 {
625 _maxID = 0;
626 Bin* ptr = _first;
627 while(ptr != nullptr)
628 {
629 std::vector<Molecule> vec = ptr->_particles;
630 auto it = max_element(vec.begin(), vec.end(), molecule_id_compare);
631 _maxID = ( (*it).getID()>_maxID ? (*it).getID() : _maxID);
632 ptr = ptr->_next;
633 if(ptr == _first)
634 break;
635 }
636 }
