199 void updateSendLeavingWithCopies(bool sendTogether){
200 // Count all processes that need to send separately
201 collCommInit(1);
202 collCommAppendInt(!sendTogether);
203 collCommAllreduceSum();
204 _sendLeavingAndCopiesSeparately = collCommGetInt();
205 collCommFinalize();
206
207
208 Log::global_log->info() << "Sending leaving particles and halo copies "
209 << (sendLeavingWithCopies() ? "together" : "separately") << std::endl;
210 }

212 bool sendLeavingWithCopies() const{
213 // No process needs to send separately => send together
214 return _sendLeavingAndCopiesSeparately == 0;
