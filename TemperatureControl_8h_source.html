213class Accumulator {
214private:
215 bool _accumulateX, _accumulateY, _accumulateZ;
216
217public:
218 Accumulator(bool accX, bool accY, bool accZ) : _accumulateX(accX), _accumulateY(accY), _accumulateZ(accZ) {}
219
220 double CalcKineticEnergyContribution(Molecule* mol) {
221 double vx = _accumulateX ? mol->v(0) : 0.0;
222 double vy = _accumulateY ? mol->v(1) : 0.0;
223 double vz = _accumulateZ ? mol->v(2) : 0.0;
224 double m = mol->mass();
225
226 return m * (vx * vx + vy * vy + vz * vz);
227 }
228 void ScaleVelocityComponents(Molecule* mol, double vcorr) {
229 if (_accumulateX) mol->setv(0, mol->v(0) * vcorr);
230 if (_accumulateY) mol->setv(1, mol->v(1) * vcorr);
231 if (_accumulateZ) mol->setv(2, mol->v(2) * vcorr);
232 }
233};
