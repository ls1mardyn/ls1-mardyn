20// Macro to wrap mardyn_exit and pass the caller file and line
21#define MARDYN_EXIT(exit_message) mardyn_exit(exit_message, __func__, __FILE__, __LINE__)
22
23inline void mardyn_exit(const std::string & exit_message,
24 const char* function, const char* filepath, const int line) {
25
26 // Only print the file path relative to the "/src/" directory
27 // The following code extracts this relative path from "filepath"
28 // Search until last "/src/" was found to avoid user-specific "src" dirs
29 const char* filepath_truncated = nullptr;
30 const char* temp = filepath;
31 while ((temp = std::strstr(temp, "/src/")) != nullptr) {
32 filepath_truncated = temp;
33 temp++;
34 }
35 if (filepath_truncated == nullptr) {
36 // Print absolute path if "/src/" not found
37 filepath_truncated = filepath;
38 } else {
39 filepath_truncated++; // +1 to ignore the first "/"
40 }
41
42 // Print code location from which MARDYN_EXIT() was called
43 Log::global_log->error_always_output()
44 << "Exit called from function `" << function << "`"
45 << " in file `" << filepath_truncated << ":" << line
46 << "` with message:" << std::endl;
47
48 // Print exit message line by line to always have Logger output
49 std::stringstream ss(exit_message);
50 std::string exit_message_line;
51 // Split exit message by "\n" and print via Logger
52 while (std::getline(ss, exit_message_line, '\n')) {
53 Log::global_log->error_always_output() << exit_message_line << std::endl;
54 }
55
56#ifdef ENABLE_MPI
57 // terminate all mpi processes and return exitcode
58 MPI_Abort(MPI_COMM_WORLD, EXIT_FAILURE);
59#else
60 // call global abort - this stops the debugger at the right spot.
61 ::abort();
62#endif
63}

65inline void __mardyn_assert__(const char * expr, const char* file, int line) {
66 std::ostringstream error_message;
67 error_message << "Assertion \"" << expr << "\" failed at " << file << ":" << line << std::endl;
68 MARDYN_EXIT(error_message.str());
69}
