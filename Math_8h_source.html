26template<typename T1, typename T2>
27inline void LinearRegression(const std::vector<T1>& x, const std::vector<T1>& y, T2& beta1, T2& beta2)
28{
29 // Reference: https://de.wikipedia.org/wiki/Lineare_Regression
30 std::vector<T1> xy;
31 typename std::vector<T1>::const_iterator it;
32 for(unsigned int i=0; i<x.size(); i++)
33 xy.push_back(x.at(i)*y.at(i) );
34
35 T2 n = (T2)(x.size() );
36 T2 invn = 1./n;
37 T2 sumx = std::accumulate(x.begin(), x.end(), 0.0);
38 T2 sumy = std::accumulate(y.begin(), y.end(), 0.0);
39 T2 sumxy = std::accumulate(xy.begin(), xy.end(), 0.0);
40 T2 sumx2 = std::accumulate(x.begin(), x.end(), 0.0, square<T2>() );
41 T2 avgx = sumx*invn;
42 T2 avgy = sumy*invn;
43
44 T2 numerator = (n*sumxy - sumx*sumy);
45 T2 dominator = (n*sumx2 - sumx*sumx);
46
47 if(dominator > 0.)
48 {
49 beta2 = numerator / dominator;
50 beta1 = avgy - beta2*avgx;
51 }
52 else
53 beta2 = beta1 = 0.;
54}
