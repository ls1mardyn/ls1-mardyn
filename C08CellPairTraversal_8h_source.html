45template<class CellTemplate, bool eighthShell>
46void C08CellPairTraversal<CellTemplate, eighthShell>::traverseCellPairs(
47 CellProcessor& cellProcessor) {
48
49 using std::array;
50 const std::array<unsigned long, 3> strides = { 2, 2, 2 };
51 std::array<unsigned long, 3> end;
52 for (int d = 0; d < 3; ++d) {
53 end[d] = this->_dims[d] - 1;
54 }
55
56 #if defined(_OPENMP)
57 #pragma omp parallel
58 #endif
59 {
60 for (unsigned long col = 0; col < 8; ++col) {
61 std::array<unsigned long, 3> begin = threeDimensionalMapping::oneToThreeD(col, strides);
62 if (eighthShell) {
63 // if we are using eighth shell, we start at 1,1,1 instead of 0,0,0
64 for (unsigned short i = 0; i < 3; ++i) {
65 begin[i] += 1;
66 }
67 }
68 traverseCellPairsBackend(cellProcessor, begin, end, strides);
69 #if defined(_OPENMP)
70 #pragma omp barrier
71 #endif
72 // this barrier is needed, since we have a nowait in the backend
73 }
74 }

77template<class CellTemplate, bool eighthShell>
78void C08CellPairTraversal<CellTemplate, eighthShell>::traverseCellPairsOuter(
79 CellProcessor& cellProcessor) {
80 if(eighthShell){
81 std::ostringstream error_message;
82 error_message << "eightshell + overlapping not yet supported." << std::endl;
83 MARDYN_EXIT(error_message.str());
84 }
85 using std::array;
86
87 {
88 unsigned long minsize = std::min(this->_dims[0], std::min(this->_dims[1], this->_dims[2]));
89
90 if (minsize <= 5) {
91 // iterating in the inner region didn't do anything. Iterate normally.
92 traverseCellPairs(cellProcessor);
93 return;
94 }
95 }
96
97 const std::array<unsigned long, 3> strides2 = { 2, 2, 2 };
98
99 #if defined(_OPENMP)
100 #pragma omp parallel
101 #endif
102 {
103 for (unsigned long col = 0; col < 8; ++col) {
104 // halo & boundaries in z direction
105 const std::array< unsigned long, 3> begin = threeDimensionalMapping::oneToThreeD(col, strides2);
106
107 // values, which are modified by 2 are actually modified by the respective stride, which is always 2
108 std::array<unsigned long, 3> startZ = { begin[0], begin[1], begin[2] }; // default
109 std::array<unsigned long, 3> endZ = { this->_dims[0] - 1, this->_dims[1] - 1, this->_dims[2] - 1 }; // default
110 std::array<unsigned long, 3> stridesZ = {strides2[0], strides2[1], this->_dims[2] - 3}; // mod z
111 traverseCellPairsBackend(cellProcessor, startZ, endZ, stridesZ);
112
113 // halo & boundaries in y direction
114 // boundaries in z direction are excluded!
115 std::array<unsigned long, 3> startY = { begin[0], begin[1], begin[2] + 2 }; // mod z
116 std::array<unsigned long, 3> endY = { this->_dims[0] - 1, this->_dims[1] - 1, this->_dims[2] - 3 }; // mod z
117 std::array<unsigned long, 3> stridesY = {strides2[0], this->_dims[1] - 3, strides2[2]}; // mod y
118 traverseCellPairsBackend(cellProcessor, startY, endY, stridesY);
119
120 // halo & boundaries in x direction
121 // boundaries in z and y direction are excluded!
122 std::array<unsigned long, 3> startX = { begin[0], begin[1] + 2, begin[2] + 2 }; // mod yz
123 std::array<unsigned long, 3> endX = { this->_dims[0] - 1, this->_dims[1] - 3, this->_dims[2] - 3 }; // mod yz
124 std::array<unsigned long, 3> stridesX = {this->_dims[0] - 3, strides2[1], strides2[2]}; // mod x
125 traverseCellPairsBackend(cellProcessor, startX, endX, stridesX);
126
127 #if defined(_OPENMP)
128 // this barrier is needed, since we have a nowait in the backend
129 // except at the last run - there we have the barrier at the end of the parallel region
130 if (col < 7) {
131 #pragma omp barrier
132 }
133 #endif
134 }
135 } // end pragma omp parallel
136}

138template<class CellTemplate, bool eighthShell>
139void C08CellPairTraversal<CellTemplate, eighthShell>::traverseCellPairsInner(
140 CellProcessor& cellProcessor, unsigned stage,
141 unsigned stageCount) {
142 using std::array;
143
144 unsigned long splitdim = 0;
145 unsigned long maxcellsize = this->_dims[0];
146 for (unsigned long i = 1; i < 3; i++){
147 if (this->_dims[i] > maxcellsize) {
148 splitdim = i;
149 maxcellsize = this->_dims[i];
150 }
151 }
152 unsigned long splitsize = maxcellsize - 5;
153 unsigned long minsize = std::min(this->_dims[0], std::min(this->_dims[1], this->_dims[2]));
154
155 mardyn_assert(minsize >= 4); // there should be at least 4 cells in each dimension, otherwise we did something stupid!
156
157 if (minsize <= 5) {
158 return; // we can not iterate over any inner cells, that do not depend on boundary or halo cells
159 }
160
161 std::array<unsigned long, 3> lower;
162 std::array<unsigned long, 3> upper;
163 for (unsigned long i = 0; i < 3; i++) {
164 lower[i] = 2;
165 upper[i] = this->_dims[i] - 3;
166 }
167 lower[splitdim] = 2 + splitsize * stage / stageCount; // at least 2
168 upper[splitdim] = 2 + splitsize * (stage+1) / stageCount; // at most _cellsPerDimension[i] - 3
169
170
171 #if defined(_OPENMP)
172 #pragma omp parallel
173 #endif
174 {
175 std::array<unsigned long, 3> strides = {2, 2, 2};
176
177 for (unsigned long col = 0; col < 8; ++col) {
178 std::array<unsigned long, 3> startIndices = threeDimensionalMapping::oneToThreeD(col, strides);
179 for (int i = 0; i < 3; i++) {
180 startIndices[i] = startIndices[i] + lower[i];
181 }
182
183 traverseCellPairsBackend(cellProcessor, startIndices, upper, strides);
184 #if defined(_OPENMP)
185 // this barrier is needed, since we have a nowait in the backend
186 // except at the last run - there we have the barrier at the end of the parallel region
187 if (col < 7) {
188 #pragma omp barrier
189 }
190 #endif
191 }
192 } // end pragma omp parallel
193}

196template<class CellTemplate, bool eighthShell>
197void C08CellPairTraversal<CellTemplate, eighthShell>::traverseCellPairsBackend(CellProcessor& cellProcessor,
198 const std::array<unsigned long, 3>& start, const std::array<unsigned long, 3>& end,
199 const std::array<unsigned long, 3>& stride) const {
200
201 // note parallel region is open outside
202
203 // intel compiler demands following:
204 // note parallel region is open outside
205 const unsigned long start_x = start[0], start_y = start[1], start_z = start[2];
206 const unsigned long end_x = end[0], end_y = end[1], end_z = end[2];
207 const unsigned long stride_x = stride[0], stride_y = stride[1], stride_z = stride[2];
208
209 // number of iterations:
210 const auto loop_size = static_cast<size_t>(std::ceil(static_cast<double>(end_x - start_x) / stride_x) *
211 std::ceil(static_cast<double>(end_y - start_y) / stride_y) *
212 std::ceil(static_cast<double>(end_z - start_z) / stride_z));
213
214 // magic numbers: empirically determined to be somewhat efficient.
215 const int chunk_size = chunk_size::getChunkSize(loop_size, 10000, 100);
216
217#if defined(_OPENMP)
218 #pragma omp for schedule(dynamic, chunk_size) collapse(3) nowait
219 #endif
220 for (unsigned long z = start_z; z < end_z; z += stride_z) {
221 for (unsigned long y = start_y; y < end_y; y += stride_y) {
222 for (unsigned long x = start_x; x < end_x; x += stride_x) {
223 unsigned long baseIndex = threeDimensionalMapping::threeToOneD(x, y, z, this->_dims);
224 C08BasedTraversals<CellTemplate>::template processBaseCell<eighthShell>(cellProcessor, baseIndex);
225 }
226 }
227 }
