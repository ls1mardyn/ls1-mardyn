234 void setOrientationVectorByPolarAngles(const double& theta_deg, const double& phi_deg) {
235 const double fac = M_PI / 180.; // translate: degrees --> rad
236 double theta_rad = theta_deg * fac;
237 double phi_rad = phi_deg * fac;
238 _e[0] = sin(theta_rad) * cos(phi_rad);
239 _e[1] = sin(theta_rad) * sin(phi_rad);
240 _e[2] = cos(theta_rad);
