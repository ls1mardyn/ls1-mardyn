239 void setOrientationVectorByPolarAngles(const double& theta_deg, const double& phi_deg) {
240 const double fac = M_PI / 180.; // translate: degrees --> rad
241 double theta_rad = theta_deg * fac;
242 double phi_rad = phi_deg * fac;
243 _e[0] = sin(theta_rad) * cos(phi_rad);
244 _e[1] = sin(theta_rad) * sin(phi_rad);
245 _e[2] = cos(theta_rad);
246 }
