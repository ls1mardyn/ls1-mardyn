104 for (int d = 0; d < 3; d++) {
105 _regionDimensions[d] = regionDimensions_arg[d];

111 _baseZ = startCellIndex_arg / (_globalDimensions[0] * _globalDimensions[1]);

113 _baseX = (startCellIndex_arg % (_globalDimensions[0] * _globalDimensions[1])) % _globalDimensions[0];

116 updateCellIteratorCell();

120 const CellContainer_T& cells = *_cells;
121 const CellIndex_T numCellsInRegion = _regionDimensions[2] * _regionDimensions[1] * _regionDimensions[0];
122
123 // if _cell_index is out of the region => invalid <=> (_localCellIndex >= numCellsInRegion)
124 if (_localCellIndex < numCellsInRegion) {
125 if (cells[_cell_index].isNotEmpty() and (this->operator*()).inBox(_startRegion, _endRegion)) {
126 // if the particle is good, leave it
127 } else {
128 // else, find a particle in the box

161inline void RegionParticleIterator :: next_non_empty_cell() {
162 //cellIndex should always be the index in the cell array (_cells member variable in LinkedCells)
163 mardyn_assert(_cells != nullptr);
164
165 const CellContainer_T& cells = *_cells;
166 const CellIndex_T numCellsInRegion = _regionDimensions[2] * _regionDimensions[1] * _regionDimensions[0];
167
168 // find the next non-empty cell
169 for (_localCellIndex += _stride; _localCellIndex < numCellsInRegion; _localCellIndex += _stride) {
170
171 const ParticleCellBase & c = cells.at(getGlobalCellIndex());
172
173 if (c.isNotEmpty() and (_type == ALL_CELLS or not c.isHaloCell())) {
174 _cell_index = getGlobalCellIndex();
175 updateCellIteratorCell();
176 break;
177 }
178 }
179}

181inline ParticleIterator::CellIndex_T RegionParticleIterator :: getGlobalCellIndex() {
182 CellIndex_T dz = _localCellIndex / (_regionDimensions[0] * _regionDimensions[1]);
183 CellIndex_T dy = (_localCellIndex % (_regionDimensions[0] * _regionDimensions[1])) / _regionDimensions[0];
184 CellIndex_T dx = (_localCellIndex % (_regionDimensions[0] * _regionDimensions[1])) % _regionDimensions[0];
185 return (_baseX + dx) + (_baseY + dy) * _globalDimensions[0] + (_baseZ + dz) * _globalDimensions[0] * _globalDimensions[1];
186}
