92 template <typename T>
93 void doTheRead(std::string var_name, std::vector<T>& container) {
94 uint64_t num = 1;
95 auto advar = io->InquireVariable<T>(var_name);
96 advar.SetStepSelection({_step, 1});
97 auto info = engine->BlocksInfo(advar, 0);
98 auto shape = info[0].Count;
99 std::for_each(shape.begin(), shape.end(), [&](decltype(num) n) { num *= n; });
100 container.resize(num);
101 engine->Get<T>(advar, container);
102 }

114 template <typename T>
115 void doTheRead(const std::string var_name, std::vector<T>& container, const uint64_t bufferSize, const uint64_t offset) {
116 adios2::Dims readsize({bufferSize});
117 adios2::Dims offs({offset});
118
119 Log::global_log->debug() << "[Adios2Reader]: Var name " << var_name << std::endl;
120 auto advar = io->InquireVariable<T>(var_name);
121 advar.SetStepSelection({_step, 1});
122 Log::global_log->debug() << "[Adios2Reader]: buffer " << bufferSize << " offset " << offset << std::endl;
123 for (auto entry : advar.Shape()) Log::global_log->debug() << "[Adios2Reader]: shape " << entry << std::endl;
124 advar.SetSelection(adios2::Box<adios2::Dims>(offs, readsize));
125 container.resize(bufferSize);
126 engine->Get<T>(advar, container);
127 }
