134void TraversalTuner<CellTemplate>::findOptimalTraversal() {
135 // TODO implement autotuning here! At the moment the traversal is chosen via readXML!
136
137 _optimalTraversal = _traversals[selectedTraversal].first;
138
139 // log traversal
140 if (dynamic_cast<HalfShellTraversal<CellTemplate> *>(_optimalTraversal))
141 Log::global_log->info() << "Using HalfShellTraversal." << std::endl;
142 else if (dynamic_cast<OriginalCellPairTraversal<CellTemplate> *>(_optimalTraversal))
143 Log::global_log->info() << "Using OriginalCellPairTraversal." << std::endl;
144 else if (dynamic_cast<C08CellPairTraversal<CellTemplate> *>(_optimalTraversal))
145 Log::global_log->info() << "Using C08CellPairTraversal without eighthShell." << std::endl;
146 else if (dynamic_cast<C08CellPairTraversal<CellTemplate, true> *>(_optimalTraversal))
147 Log::global_log->info() << "Using C08CellPairTraversal with eighthShell." << std::endl;
148 else if (dynamic_cast<C04CellPairTraversal<CellTemplate> *>(_optimalTraversal))
149 Log::global_log->info() << "Using C04CellPairTraversal." << std::endl;
150 else if (dynamic_cast<MidpointTraversal<CellTemplate> *>(_optimalTraversal))
151 Log::global_log->info() << "Using MidpointTraversal." << std::endl;
152 else if (dynamic_cast<NeutralTerritoryTraversal<CellTemplate> *>(_optimalTraversal))
153 Log::global_log->info() << "Using NeutralTerritoryTraversal." << std::endl;
154 else if (dynamic_cast<QuickschedTraversal<CellTemplate> *>(_optimalTraversal)) {
155 Log::global_log->info() << "Using QuickschedTraversal." << std::endl;
156#ifndef QUICKSCHED
157 std::ostringstream error_message;
158 error_message << "MarDyn was compiled without Quicksched Support. Aborting!" << std::endl;
159 MARDYN_EXIT(error_message.str());

161 } else if (dynamic_cast<SlicedCellPairTraversal<CellTemplate> *>(_optimalTraversal))
162 Log::global_log->info() << "Using SlicedCellPairTraversal." << std::endl;
163 else
164 Log::global_log->warning() << "Using unknown traversal." << std::endl;
165
166 if (_cellsInCutoff > _optimalTraversal->maxCellsInCutoff()) {
167 std::ostringstream error_message;
168 error_message << "Traversal supports up to " << _optimalTraversal->maxCellsInCutoff()
169 << " cells in cutoff, but value is chosen as " << _cellsInCutoff << std::endl;
170 MARDYN_EXIT(error_message.str());

174template<class CellTemplate>
175void TraversalTuner<CellTemplate>::readXML(XMLfileUnits &xmlconfig) {
176 std::string oldPath(xmlconfig.getcurrentnodepath());
177 // read traversal type default values
178 std::string traversalType;
179
180 xmlconfig.getNodeValue("traversalSelector", traversalType);
181 transform(traversalType.begin(), traversalType.end(), traversalType.begin(), ::tolower);
182
183 if (traversalType.find("c08es") != std::string::npos)
184 selectedTraversal = C08ES;
185 else if (traversalType.find("c08") != std::string::npos)
186 selectedTraversal = C08;
187 else if (traversalType.find("c04") != std::string::npos)
188 selectedTraversal = C04;
189 else if (traversalType.find("qui") != std::string::npos)
190 selectedTraversal = QSCHED;
191 else if (traversalType.find("slice") != std::string::npos)
192 selectedTraversal = SLICED;
193 else if (traversalType.find("ori") != std::string::npos)
194 selectedTraversal = ORIGINAL;
195 else if (traversalType.find("hs") != std::string::npos)
196 selectedTraversal = HS;
197 else if (traversalType.find("mp") != std::string::npos)
198 selectedTraversal = MP;
199 else if (traversalType.find("nt") != std::string::npos) {
200 selectedTraversal = NT;
201 } else {
202 // selector already set in constructor, just print a warning here
203 if (mardyn_get_max_threads() > 1) {
204 Log::global_log->warning() << "No traversal type selected. Defaulting to c08 traversal." << std::endl;
205 } else {
206 Log::global_log->warning() << "No traversal type selected. Defaulting to sliced traversal." << std::endl;
207 }
208 }
209
210 _cellsInCutoff = xmlconfig.getNodeValue_int("cellsInCutoffRadius", 1); // This is currently only used for an assert
211
212 // workaround for stupid iterator:
213 // since
214 // xmlconfig.changecurrentnode(traversalIterator);
215 // does not work resolve paths to traversals manually
216 // use iterator only to resolve number of traversals (==iterations)
217 std::string basePath(xmlconfig.getcurrentnodepath());
218
219 int i = 1;
220 XMLfile::Query qry = xmlconfig.query("traversalData");
221 for (XMLfile::Query::const_iterator traversalIterator = qry.begin(); traversalIterator; ++traversalIterator) {
222 std::string path(basePath + "/traversalData[" + std::to_string(i) + "]");
223 xmlconfig.changecurrentnode(path);
224
225 traversalType = xmlconfig.getNodeValue_string("@type", "NOTHING FOUND");
226 transform(traversalType.begin(), traversalType.end(), traversalType.begin(), ::tolower);
227 if (traversalType == "c08") {
228 // nothing to do
229 } else if (traversalType.find("qui") != std::string::npos) {
230#ifdef QUICKSCHED
231 if (not std::is_base_of<ParticleCellBase, CellTemplate>::value) {
232 Log::global_log->warning() << "Attempting to use Quicksched with cell type that does not store task data!"
233 << std::endl;
234 }
235 for (auto p : _traversals) {
236 if (struct QuickschedTraversalData *quiData = dynamic_cast<QuickschedTraversalData *>(p.second)) {
237 // read task block size
238 std::string tag = "taskBlockSize/l";
239 char dimension = 'x';
240
241 for (int j = 0; j < 3; ++j) {
242 tag += (dimension + j);
243 xmlconfig.getNodeValue(tag, quiData->taskBlockSize[j]);
244 if (quiData->taskBlockSize[j] < 2) {
245 std::ostringstream error_message;
246 error_message << "Task block size in "
247 << (char) (dimension + j)
248 << " direction is <2 and thereby invalid! ("
249 << quiData->taskBlockSize[j] << ")"
250 << std::endl;
251 MARDYN_EXIT(error_message.str());
252 }
253 }
254 break;
255 }
256 }
257#else
258 Log::global_log->warning() << "Found quicksched traversal data in config "
259 << "but mardyn was compiled without quicksched support! "
260 << "(make ENABLE_QUICKSCHED=1)" << std::endl;
261#endif
262 } else {
263 Log::global_log->warning() << "Unknown traversal type: " << traversalType << std::endl;
264 }
265 ++i;
266 }
267 xmlconfig.changecurrentnode(oldPath);
268}

270template<class CellTemplate>
271void TraversalTuner<CellTemplate>::rebuild(std::vector<CellTemplate> &cells, const std::array<unsigned long, 3> &dims,
272 double cellLength[3], double cutoff) {
273 _cells = &cells; // new - what for?
274 _dims = dims; // new - what for?
275
276 for (size_t i = 0ul; i < _traversals.size(); ++i) {
277 auto& [traversalPointerReference, traversalData] = _traversals[i];
278 // decide whether to initialize or rebuild
279 if (traversalPointerReference == nullptr) {
280 switch (i) {
281 case traversalNames ::ORIGINAL:
282 traversalPointerReference = new OriginalCellPairTraversal<CellTemplate>(cells, dims);
283 break;
284 case traversalNames::C08:
285 traversalPointerReference = new C08CellPairTraversal<CellTemplate>(cells, dims);
286 break;
287 case traversalNames::C04:
288 traversalPointerReference = new C04CellPairTraversal<CellTemplate>(cells, dims);
289 break;
290 case traversalNames::SLICED:
291 traversalPointerReference = new SlicedCellPairTraversal<CellTemplate>(cells, dims);
292 break;
293 case traversalNames::HS:
294 traversalPointerReference = new HalfShellTraversal<CellTemplate>(cells, dims);
295 break;
296 case traversalNames::MP:
297 traversalPointerReference = new MidpointTraversal<CellTemplate>(cells, dims);
298 break;
299 case traversalNames::NT:
300 traversalPointerReference =
301 new NeutralTerritoryTraversal<CellTemplate>(cells, dims, cellLength, cutoff);
302 break;
303 case traversalNames::C08ES:
304 traversalPointerReference = new C08CellPairTraversal<CellTemplate, true>(cells, dims);
305 break;
306 case traversalNames::QSCHED: {
307 mardyn_assert((std::is_base_of<ParticleCellBase, CellTemplate>::value));
308 auto *quiData = dynamic_cast<QuickschedTraversalData *>(traversalData);
309 traversalPointerReference = new QuickschedTraversal<CellTemplate>(cells, dims, quiData->taskBlockSize);
310 } break;
311 default:
312 std::ostringstream error_message;
313 error_message << "Unknown traversal data found in TraversalTuner._traversals!" << std::endl;
314 MARDYN_EXIT(error_message.str());
315 }
316 }
317 traversalPointerReference->rebuild(cells, dims, cellLength, cutoff, traversalData);
318 }
319 _optimalTraversal = nullptr;
320}

331inline void TraversalTuner<CellTemplate>::traverseCellPairs(traversalNames name,
332 CellProcessor& cellProcessor) {
333 if (name == getSelectedTraversal()) {
334 traverseCellPairs(cellProcessor);
335 } else {
336 SlicedCellPairTraversal<CellTemplate> slicedTraversal(*_cells, _dims);
337 switch(name) {
338 case SLICED:
339 slicedTraversal.traverseCellPairs(cellProcessor);
340 break;
341 default:
342 std::ostringstream error_message;
343 error_message<< "Calling traverseCellPairs(traversalName, CellProcessor&) for something else than the Sliced Traversal is disabled for now. Aborting." << std::endl;
344 MARDYN_EXIT(error_message.str());
345 break;
346 }
347 }
